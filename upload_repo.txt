#!/bin/bash
# ===============================
# Upload Fresh Folder + index.html ke GitHub Pages
# ===============================

# 1️⃣ Masuk folder project lokal (ganti path sesuai folder lo)
cd C:Users/LENOVO/Documents/har-website/result

# 2️⃣ Hapus semua file/folder di repo lokal agar bersih
# WARNING: Pastikan ini folder project yang benar
git rm -r *
git commit -m "Hapus semua file lama"

# 3️⃣ Tambahkan file/folder yang mau di-upload saja
git add index.html
git add result/

# 4️⃣ Commit file baru
git commit -m "Upload folder result + index.html fresh"

# 5️⃣ Pastikan branch utama bernama main
git branch -M main

# 6️⃣ Set remote repo GitHub (ganti URL sesuai repo lo)
git remote set-url origin https://github.com/yuddiprattama-eng/file.git

# 7️⃣ Force push ke GitHub (overwrite remote)
git push -u origin main --force

# ✅ Selesai
echo "Upload selesai! Tunggu beberapa menit lalu cek:"
echo "https://yuddiprattama-eng.github.io/echa_rangga/"
